%% ================================================================ DISCUSSION
\headerslide{Discussion}

%% ---------------------------------------------------------------- caveats
\begin{frame}{What we are not saying}
  \begin{itemize}[<+->]
    \item Not that LLMs are bad, or that they have not learned at all
    \item Not that ICL is useless, or that it cannot learn anything
    \item Not that Transformers can't learn \textsc{$n$-Gram}s, or even that they can't learn DFAs
    \item Not that LLMs know nothing about language, or that they can't learn any languages in-context
  \end{itemize}
\end{frame}

%% ---------------------------------------------------------------- the claim
%% One sentence, revealed clause by clause (each clause on its own line).
\begin{frame}{But we can say this!}
  \large
  \onslide<1->{LLMs have not learned}\\
  \onslide<2->{\hspace{1.5em}a \hl{data-efficient}}\\
  \onslide<3->{\hspace{1.5em}and \hl{general purpose} learning algorithm}\\
  \onslide<4->{\hspace{1.5em}that they can apply \hl{in-context}}\\
  \onslide<5->{\hspace{1.5em}to an \hl{entirely new} language}\\
  \onslide<6->{\hspace{1.5em}\hl{without knowing anything} about it.}\\[0.7em]
  \onslide<7->{Even when}\\
  \onslide<8->{\hspace{1.5em}languages are \hl{regular}}\\
  \onslide<9->{\hspace{1.5em}and the only target to beat is an \textsc{$n$-Gram}}\\
  \onslide<10->{\hspace{1.5em}with \hl{no ability to world model}.}
\end{frame}
