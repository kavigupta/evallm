%% ---------------------------------------------------------------- the ideal task
\begin{frame}{The ideal task}
  A task that isolates ICL needs three properties:
  \vspace{0.6em}
  \begin{enumerate}[<+->]
    \item \hl{A language task}: LLMs are first and foremost language models,
          so this is their fairest arena.
    \item \hl{Drawn at random}: the process of hand writing a task inherently
          and unavoidably is biased towards the canonical
          \begin{itemize}[<+->]
            \item In particular, we want to draw from a \hl{parsimoniously defined}
                space of languages, to avoid any bias whatsoever
          \end{itemize}
    \item \hl{Baselines that do not model the world}: to isolate the effect of novelty, we need
          baselines that have no world model
  \end{enumerate}
  \vfill
  \onslide<+->{\hl{Our answer:} regular languages languages recognized by $k$-state DFAs.}
\end{frame}

%% ---------------------------------------------------------------- why 3-state DFAs
\begin{frame}{Why 3-state DFAs?}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}[<+->]
        \item Regular languages are the \hl{simplest} class in the Chomsky hierarchy
        \item Difficulty $=$ \hl{number of states}: the working memory a model
              must maintain.
        \item \hl{2-state} DFAs have \hl{no hidden state} (states $=$ outputs
              $\{0,1\}$)
        \item \hl{3-state} DFAs are the \hl{simplest nontrivial} case
        \item Caveat: only \hl{78{,}786} of them. No mathematical guarantee of novelty
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \centering
      \resizebox{0.95\linewidth}{!}{%
      \begin{tikzpicture}[
          >=Stealth, node distance=26mm,
          state/.style={circle, draw=evallmText, line width=1pt,
            minimum size=11mm, font=\large},
          accept/.style={state, double, double distance=1pt},
          every edge/.style={draw, line width=1pt}]
        \node[accept] (s0) at (90:24mm) {0};
        \node[state]  (s1) at (210:24mm) {1};
        \node[state]  (s2) at (-30:24mm) {2};

        %% self-loops on a (blue): s+0 = s
        \draw[ecblue, ->] (s0) edge[loop above] node[above]{\symA} (s0);
        \draw[ecblue, ->] (s1) edge[loop left]  node[left]{\symA}  (s1);
        \draw[ecblue, ->] (s2) edge[loop right] node[right]{\symA} (s2);

        %% b (orange): +1 -> 0->1->2->0
        \draw[ecorange, ->] (s0) to[bend left=16] node[auto, inner sep=2pt]{\symB} (s1);
        \draw[ecorange, ->] (s1) to[bend left=16] node[auto, inner sep=2pt]{\symB} (s2);
        \draw[ecorange, ->] (s2) to[bend left=16] node[auto, inner sep=2pt]{\symB} (s0);

        %% c (purple): +2 -> 0->2->1->0.  Each c edge is anti-parallel to a b
        %% edge on the same side; using the SAME bend direction makes the two
        %% arcs bow to opposite physical sides, so the auto-placed labels land
        %% on opposite sides and never overlap.
        \draw[ecpurple, ->] (s0) to[bend left=16] node[auto, inner sep=2pt]{\symC} (s2);
        \draw[ecpurple, ->] (s2) to[bend left=16] node[auto, inner sep=2pt]{\symC} (s1);
        \draw[ecpurple, ->] (s1) to[bend left=16] node[auto, inner sep=2pt]{\symC} (s0);
      \end{tikzpicture}}
      \par\vspace{0.7em}
      {\scriptsize A 3-state DFA: ``sum of symbols (\symA\,$=0$, \symB\,$=1$,
       \symC\,$=2$) divisible by 3.'' State \texttt{0} (double ring) accepts.}
    \end{column}
  \end{columns}
\end{frame}

%% ================================================================ BASELINES
\headerslide{Baselines}

%% ---------------------------------------------------------------- random/null
\begin{frame}{Baseline 1: \textsc{Random} / \textsc{Null}}
  \begin{itemize}[<+->]
    \item \hl{\textsc{Random}}: on sequence completion, predict a
          \hl{random completion}.
    \item \hl{\textsc{Null}}: on the transducer, pick whichever of \hl{0} and
          \hl{1} is better.
    \item These \hl{ignore the examples / context} entirely.
    \item Establish a \hl{chance baseline}.
  \end{itemize}
\end{frame}

%% ---------------------------------------------------------------- common-suffix
\begin{frame}{Baseline 2: \textsc{Common-Suffix}}
  \begin{columns}[T]
    \begin{column}{0.55\textwidth}
      \begin{itemize}[<+->]
        \item A heuristic for the \hl{sequence-completion} task.
        \item \hl{Ignores the prefix}; just appends a suffix that the examples
              tend to end with.
        \item Picks the suffix $S$ maximizing
              $\bigl(\#\text{examples ending in } S\bigr)\times|S|$, trading
              frequency against length.
        \item Pure \hl{surface copying}: no notion of state, no real inference.
      \end{itemize}
    \end{column}
    \begin{column}{0.45\textwidth}
      \centering
      \onslide<2->{%
      \ttfamily\large
      \begin{tabular}{l}
        \symB\,\symB\,\symC\,\symC\,\hl{\symC\,\symB\,\symA}\\
        \symA\,\symC\,\symB\,\symB\,\hl{\symC\,\symB\,\symA}\\
        \symB\,\symA\,\symA\,\symC\,\hl{\symC\,\symB\,\symA}\\[6pt]
        \symC\,\symB\,\symA\,?\,?\,?\,?\\
      \end{tabular}}
      \par\vspace{0.8em}
      \onslide<3->{%
      {\normalfont\footnotesize All 3 end in ``\,\symC\,\symB\,\symA\,''
      ($3\times3=9$), beating ``\,\symC\,\symC\,\symB\,\symA\,'' ($2\times4=8$).}}
    \end{column}
  \end{columns}
\end{frame}

\begin{frame}{\textsc{$n$-Gram} on Transducer tasks}
  \begin{center}
  \ttfamily\normalsize
  \begin{tabular}{@{}l@{}}
    \symB\,\texttt{0}\,\symC\,\texttt{1}\,\symB\,\texttt{0}\,\symC\,\texttt{1}\,\symA\,\texttt{1}\,\symB\,\texttt{0}\,\symB\,\texttt{0}\,\symB\,\texttt{1}\,\symC\,\texttt{0}\,\symA\,\texttt{0}\\[0.7em]
    \symA\,\texttt{0}\,\symC\,\alt<3->{\mt{\texttt{0}\,\symA\,\texttt{0}\,\symB}}{\texttt{0}\,\symA\,\texttt{0}\,\symB}\,\alt<4->{\pulled{\texttt{0}}}{\texttt{0}}\,\symC\,\texttt{0}\,\symC\,\texttt{1}\,\symB\,\alt<3->{\mt{\texttt{0}\,\symA\,\texttt{0}\,\symB}}{\texttt{0}\,\symA\,\texttt{0}\,\symB}\,\alt<4->{\pulled{\texttt{0}}}{\texttt{0}}\,\symC\,\texttt{0}\\[0.7em]
    \symB\,\alt<3->{\mt{\texttt{0}\,\symA\,\texttt{0}\,\symB}}{\texttt{0}\,\symA\,\texttt{0}\,\symB}\,\alt<4->{\pulled{\texttt{1}}}{\texttt{1}}\,\symB\,\texttt{0}\,\symC\,\texttt{1}\,\symA\,\texttt{1}\,\symB\,\alt<2->{\qt{\texttt{0}\,\symA\,\texttt{0}\,\symB}}{\texttt{0}\,\symA\,\texttt{0}\,\symB}\,\mask
  \end{tabular}
  \end{center}
  \vspace{0.6em}
  \small
  \onslide<1->{Lay out the input/output as interleaved sequences.}\par\medskip
  \onslide<2->{\textbf{Context:} the last $n{-}1=4$ tokens,
  \qt{\texttt{0}\,\symA\,\texttt{0}\,\symB}.}\par\smallskip
  \onslide<3->{Find every \textcolor{ecblue}{earlier occurrence} of that context\only<4->{, and take the \textcolor{ecgreen}{output that follows} each.}}\par\smallskip
  \onslide<5->{Counts: \texttt{0} appears \textcolor{ecgreen}{twice}, \texttt{1} \textcolor{ecgreen}{once}.}\par\smallskip
  \onslide<6->{Majority vote $\Rightarrow$ \textcolor{ecgreen}{0} \yes}
\end{frame}

%% ---------------------------------------------------------------- brute force
%% Colored equation terms, each revealed/highlighted on its own overlay beat.
\newcommand{\cmark}{\text{\ding{51}}}
%% \hlm: a content-fitted highlight box for math atoms (lua-ul's rule height is
%% fixed and mis-sizes sub/superscripts and large operators).
\newcommand{\hlm}[2]{\tikz[baseline=(hh.base)]{\node[fill=#1, inner sep=1.2pt, outer sep=0pt, rounded corners=1.5pt] (hh) {\ensuremath{#2}};}}
%% \hlmo: pick between a filled box (from overlay #1) and an identically-sized
%% unfilled box (before). Same dimensions either way, so no reflow/jitter.
\newcommand{\hlmo}[3]{\alt<#1>{\hlm{#2}{#3}}{\hlm{none}{#3}}}
\newcommand{\bx}{\hlmo{2-}{ecblue!18}{x}}
\newcommand{\byh}{\hlmo{3-}{ecpurple!20}{\hat y}}
\newcommand{\byhsub}{\hlmo{3-}{ecpurple!20}{\scriptstyle\hat y}}
\newcommand{\bcm}{\hlmo{4-}{ecgreen!28}{\cmark}}
\newcommand{\bsum}{\hlmo{6-}{evallmHighlight!30}{\displaystyle\sum_{\mathrm{dfa}}}}
\begin{frame}{Baseline 4: \textsc{Brute-Force}}
  \onslide<1->{We pick the prediction that \hl{maximizes the chance of being
  correct}:}
  \vspace{0.6em}
  \[
  \begin{gathered}
    \operatorname*{arg\,max}_{\byhsub}\; P\!\left(\bcm \;\middle|\; \bx ;\, \byh\right)\\[8pt]
    \onslide<5->{P\!\left(\cmark \;\middle|\; x ;\, \hat y\right) =
      \bsum\; P\!\left(\cmark \;\middle|\; x,\mathrm{dfa};\, \hat y\right)\,P(\mathrm{dfa})}
  \end{gathered}
  \]
  \vspace{0.3em}
  \small
  \onslide<2->{\hlm{ecblue!18}{x}: the input/output examples.}\par\smallskip
  \onslide<3->{\hlm{ecpurple!20}{\hat y}: the predicted output.}\par\smallskip
  \onslide<4->{\hlm{ecgreen!28}{\cmark}: whether the prediction is correct.}\par\smallskip
  \onslide<5->{Marginalize over every DFA, with $P(\mathrm{dfa})$ taken \hl{uniform}.}\par\smallskip
  \onslide<6->{The \hlm{evallmHighlight!30}{\textstyle\sum_{\mathrm{dfa}}} is
  tractable: there are only \hl{78{,}786} 3-state DFAs.}
\end{frame}

%% ---------------------------------------------------------------- summary
\begin{frame}{Baselines: a spectrum}
  \begin{itemize}[<+->]
    \item \hl{\textsc{Random} / \textsc{Null}}: a \hl{lower bound} (chance,
          no learning at all).
    \item \hl{\textsc{Common-Suffix}} and \hl{\textsc{$n$-Gram}}:
          \hl{non world-modeling} baselines of varying sophistication.
    \item \hl{\textsc{Brute-Force}}: an \hl{upper bound} (the Bayes-optimal
          world model).
  \end{itemize}
\end{frame}

%% ================================================================ DISCUSSION
\headerslide{Discussion}

%% ---------------------------------------------------------------- caveats
\begin{frame}{What we are not saying}
  \begin{itemize}[<+->]
    \item Not that LLMs are bad, or that they have not learned at all
    \item Not that ICL is useless, or that it cannot learn anything
    \item Not that Transformers can't learn \textsc{$n$-Gram}s, or even that they can't learn DFAs
    \item Not that LLMs know nothing about language, or that they can't learn any languages in-context
  \end{itemize}
\end{frame}

%% ---------------------------------------------------------------- caveats
\begin{frame}{But we can say this}
  \begin{itemize}[<+->]
    \item LLMs have not learned a \hl{general theory of language} in their \hl{parameters} that they can apply \hl{in-context} to an \hl{entirely new} language \hl{without knowing anything} about it.
  \end{itemize}
\end{frame}
